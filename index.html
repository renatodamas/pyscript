Cálculo de hipotenusa de um triângulo retângulo


from math import sqrt def hypotenuse(b, c): return sqrt(b**2 + c**2) def hypotenuse_result(): b = float(Element("opposite").value) c = float(Element("adjacent").value) Element("hypotenuse-result").write(f"comprimento da hipotenusa é {hipotenusa(b, c):.2f}") def clean(): Element("opposite").clear() Element("adjacent").clear() Element("hypotenuse-result").clear()
